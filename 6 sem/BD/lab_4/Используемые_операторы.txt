2) WHERE (покупатель, товар)
3) LIKE (сотрудник, склад)
4) ORDER BY ASC (импортер, склад)
5) ORDER BY DESС (покупатель, сотрудник)
6) BETWEEN AND (склад, товар)
7) AND (импортер, магазин)
8) IN (товар, склад)
9) GROUP BY (магазин, сотрудник)
10) NULLS FIRST (импортер, покупатель)
11) NULLS LAST (товар, сотрудник)
12) OFFSET (магазин, импортер)
13) LIMIT (склад, покупатель)
14) JOIN (сотрудник, импортер)
15) DISTINCT (магазин, товар)
